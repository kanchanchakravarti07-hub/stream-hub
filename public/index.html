import express from 'express'; import fs from 'fs'; import path from 'path'; import { fileURLToPath } from 'url'; const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); const app = express(); // Public folder serve karo app.use(express.static('public')); // Channels API (Sahi path ke saath) app.get('/channels', (req, res) => { try { const filePath = path.join(__dirname, 'channels.json'); const data = fs.readFileSync(filePath, 'utf8'); res.json(JSON.parse(data)); } catch (err) { console.error("Error reading channels.json:", err); res.status(500).json({ error: "File not found" }); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
